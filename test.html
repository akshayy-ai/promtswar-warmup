describe('parseMeta', () => {
  const sample = `Some text
`;

  it('returns null when no META block present', () => {
    expect(parseMeta('no metadata here')).toBe(null);
  });
  it('parses level correctly', () => {
    expect(parseMeta(sample).level).toBe('intermediate');
  });
  it('parses understanding as integer', () => {
    expect(parseMeta(sample).understanding).toBe(65);
  });
  it('parses concepts_mastered', () => {
    expect(parseMeta(sample).concepts_mastered).toBe('variables, functions');
  });
  it('parses concepts_struggling', () => {
    expect(parseMeta(sample).concepts_struggling).toBe('closures');
  });
  it('understanding defaults to 0 for invalid value', () => {
    const t = '';
    expect(parseMeta(t).understanding).toBe(0);
  });
});

describe('stripMeta', () => {
  it('removes META block from response', () => {
    const text = 'Hello world\n';
    expect(stripMeta(text)).toBe('Hello world');
  });
  it('leaves text without META unchanged', () => {
    expect(stripMeta('plain text')).toBe('plain text');
  });
  it('trims surrounding whitespace', () => {
    expect(stripMeta('  hello    ')).toBe('hello');
  });
});

describe('Security', () => {
  it('XSS in topic input is escaped by escapeHtml', () => {
    const malicious = '';
    expect(escapeHtml(malicious)).toContain('<img');
  });
  it('script tag is fully neutralised', () => {
    const out = escapeHtml('');
    expect(out).toContain('<script>');
  });
  it('code block content is escaped to prevent XSS', () => {
    const out = parseMarkdown('```\n\n```');
    expect(out).toContain('<script>');
  });
});

describe('Understanding score bounds', () => {

  });
  it('score of 100 is valid upper bound', () => {
    const meta = '';
    expect(parseMeta(meta).understanding).toBeLessThanOrEqual(100);
  });
});

const output = document.getElementById('output');
let passed = 0, failed = 0;

results.forEach(suite => {
  const div = document.createElement('div');
  div.className = 'suite';
  div.innerHTML = `
    ${suite.name}
    `;
  suite.tests.forEach(t => {
    const line = document.createElement('div');
    if (t.passed) {
      line.className = 'pass';
      line.textContent = `  ✓ ${t.label}`;
      passed++;
    } else {
      line.className = 'fail';
      line.textContent = `  ✗ ${t.label} — ${t.error}`;
      failed++;
    }
    div.appendChild(line);
  });
  output.appendChild(div);
});

const summary = document.getElementById('summary');
const total = passed + failed;
summary.innerHTML = failed === 0
  ? `✓ All ${total} tests passed`
  : `✗ ${failed} failed / ${passed} passed (${total} total)`;
